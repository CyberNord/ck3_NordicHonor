			AND = {

				age >= 18

		trigger_event = {
			id = nh.0001
			days = {5 10}
		}
	}
	
	ai_potential = {
		highest_held_title_tier > tier_barony
		is_landed = yes
		faith = faith:norse_pagan
	}

	ai_will_do = {
		base = 10

		modifier = {
            add = 70
            has_trait = faith_warrior

		modifier = {
            add = 30
            has_trait = brave
        }

		modifier = {
            add = 15
            has_trait = zealous

		modifier = {
            add = 15
            has_trait = holy_warrior
        }

		modifier = {
            add = 10

        }

            add = -100
            has_trait = craven
        }
	}
